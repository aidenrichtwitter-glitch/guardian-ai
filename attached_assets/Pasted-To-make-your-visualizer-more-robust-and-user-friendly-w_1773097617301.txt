To make your visualizer more **robust** and user-friendly, we'll address these key improvements:

- Switch songs easily: Choose a new file anytime → old one stops, new one loads and plays
- Pause/resume support (button toggles Play/Pause)

  <script>
    const audioFileInput = document.getElementById('audioFile');
    const fileNameSpan = document.getElementById('fileName');
    const playPauseBtn = document.getElementById('playPauseBtn');
    const stopBtn = document.getElementById('stopBtn');
    const canvas = document.getElementById('visualizer');
    const statusDiv = document.getElementById('status');

    const ctx = canvas.getContext('2d');
    let audioContext = null;
    let analyser = null;
    let source = null;
    let audioElement = null;
    let isPlaying = false;
    let animationFrameId = null;

    // Restore last file name and time from localStorage (position only, file must be re-selected)
    let lastFileName = localStorage.getItem('lastAudioFileName') || null;
    let lastTime = parseFloat(localStorage.getItem('lastAudioTime')) || 0;

    if (lastFileName) {
      fileNameSpan.textContent = lastFileName + ' (previously played)';
      statusDiv.textContent = 'Previous track detected – select it again to resume';
    }

    // Create hidden <audio> element for reliable playback
    audioElement = document.createElement('audio');
    audioElement.crossOrigin = 'anonymous';
    document.body.appendChild(audioElement);

    // File selection handler
    audioFileInput.addEventListener('change', (e) => {
      const file = e.target.files[0];
      if (!file) return;

      fileNameSpan.textContent = file.name;
      statusDiv.textContent = 'File selected – ready to play';

      const url = URL.createObjectURL(file);
      audioElement.src = url;

      // Save file name (but not the file itself – security restriction)
      localStorage.setItem('lastAudioFileName', file.name);

      // Resume from last time if same file name
      if (file.name === lastFileName) {
        audioElement.currentTime = lastTime;
        statusDiv.textContent += ` (resuming from ${Math.floor(lastTime)}s)`;
      }

      playPauseBtn.disabled = false;
      stopBtn.disabled = false;

      initAudioContext(); // Prepare analyser once we have a source
    });

    // Initialize Web Audio API on first user interaction (browser policy)
    function initAudioContext() {
      if (audioContext) return;

      audioContext = new (window.AudioContext || window.webkitAudioContext)();
      analyser = audioContext.createAnalyser();
      analyser.fftSize = 256; // For bars – smaller = smoother, larger = more detail
      const bufferLength = analyser.frequencyBinCount;
      const dataArray = new Uint8Array(bufferLength);

      source = audioContext.createMediaElementSource(audioElement);
      source.connect(analyser);
      analyser.connect(audioContext.destination);

      // Start visualization loop once context is ready
      drawVisualizer();
    }

    // Play/Pause toggle
    playPauseBtn.addEventListener('click', () => {
      if (!audioElement.src) {
        statusDiv.textContent = 'No file selected';
        return;
      }

      if (audioContext?.state === 'suspended') {
        audioContext.resume().then(() => {
          togglePlay();
        });
      } else {
        togglePlay();
      }
    });

    function togglePlay() {
      if (isPlaying) {
        audioElement.pause();
        playPauseBtn.textContent = 'Play';
        statusDiv.textContent = 'Paused';
      } else {
        audioElement.play().then(() => {
          playPauseBtn.textContent = 'Pause';
          statusDiv.textContent = 'Playing';
        }).catch(err => {
          statusDiv.textContent = 'Play failed: ' + err.message;
          console.error(err);
        });
      }
      isPlaying = !isPlaying;
    }

    // Stop button – reset to start
    stopBtn.addEventListener('click', () => {
      audioElement.pause();
      audioElement.currentTime = 0;
      playPauseBtn.textContent = 'Play';
      isPlaying = false;
      statusDiv.textContent = 'Stopped';
    });

    // Save current time every few seconds
    setInterval(() => {
      if (isPlaying && audioElement.currentTime > 0) {
        localStorage.setItem('lastAudioTime', audioElement.currentTime);
      }
    }, 3000);

    // Basic bar visualizer
    function drawVisualizer() {
      animationFrameId = requestAnimationFrame(drawVisualizer);

      analyser.getByteFrequencyData(dataArray);
      ctx.fillStyle = 'rgba(0,0,0,0.2)';
      ctx.fillRect(0, 0, canvas.width, canvas.height);

      const barWidth = (canvas.width / dataArray.length) * 2.5;
      let x = 0;

      for (let i = 0; i < dataArray.length; i++) {
        const barHeight = (dataArray[i] / 255) * canvas.height * 0.8;

        const hue = i * 2 + 180; // Purple-blue range
        ctx.fillStyle = `hsl(${hue}, 80%, 60%)`;

        ctx.fillRect(x, canvas.height - barHeight, barWidth, barHeight);
        x += barWidth + 1;
      }
    }

    // Cleanup on unload
    window.addEventListener('beforeunload', () => {
      if (animationFrameId) cancelAnimationFrame(animationFrameId);
      if (audioElement) audioElement.pause();
    });
  </script>
</body>

### How to Test & Use

2. Open in browser.
3. Click "Choose File" → pick an MP3/WAV.
4. Click "Play" → visualizer starts, button becomes "Pause".
5. Switch songs: Just choose a new file → old stops, new starts.
6. Leave page & return: Re-select the same file name → it resumes from last position (if browser kept the session).
7. Pause/Stop as needed.

If the button feels "slow" → it's usually the first-time AudioContext resume + file decoding. Subsequent plays are instant.
